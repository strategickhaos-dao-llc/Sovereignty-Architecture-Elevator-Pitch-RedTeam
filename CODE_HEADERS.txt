# Function to prepend header to file
prepend_header() {
    local file=$1
    local header=$2
    
    # Create backup
    cp "$file" "$file.bak"
    
    # Check if header already exists
    if grep -q "Strategickhaos DAO LLC" "$file"; then
        echo "Header already exists in $file, skipping..."
        rm "$file.bak"
        return
    fi
    
    # Prepend header
    echo "$header" | cat - "$file" > temp && mv temp "$file"
    echo "Added header to $file"
    rm "$file.bak"
}

# Apply to Python files
echo "Applying headers to Python files..."
find . -name "*.py" -type f | while read file; do
    prepend_header "$file" "$HEADER_PYTHON"
done

# Apply to JavaScript/TypeScript files
echo "Applying headers to JavaScript/TypeScript files..."
find . -name "*.js" -o -name "*.ts" -o -name "*.jsx" -o -name "*.tsx" -type f | while read file; do
    prepend_header "$file" "$HEADER_JS"
done

# Apply to Bash scripts (excluding this script)
echo "Applying headers to Bash scripts..."
find . -name "*.sh" -type f ! -name "apply_headers.sh" | while read file; do
    # Remove shebang if present, we'll add it in header
    if head -n 1 "$file" | grep -q "^#!/bin/bash"; then
        tail -n +2 "$file" > temp && mv temp "$file"
    fi
    prepend_header "$file" "$HEADER_BASH"
done

echo "Header application complete!"

$HEADER_JS = @'
/**
 * Copyright (c) 2025 Strategickhaos DAO LLC
 * Licensed under the Strategickhaos Charitable Covenant License v1.0
 * Patent Pending - All patent rights reserved
 * 
 * This file is part of the Strategickhaos Sovereignty Architecture.
 * 
 * Non-commercial use only. Commercial use requires a separate license.
 * The mandatory 7% charitable allocation mechanism must remain intact.
 * 
 * For commercial licensing: licensing@strategickhaos.org
 * Full license: https://github.com/Strategickhaos/Sovereignty-Architecture-Elevator-Pitch-/blob/main/LICENSE
 */

'@

function Add-Header {
    param(
        [string]$FilePath,
        [string]$Header
    )
    
    $content = Get-Content $FilePath -Raw
    
    if ($content -match "Strategickhaos DAO LLC") {
        Write-Host "Header already exists in $FilePath, skipping..." -ForegroundColor Yellow
        return
    }
    
    $newContent = $Header + $content
    Set-Content -Path $FilePath -Value $newContent
    Write-Host "Added header to $FilePath" -ForegroundColor Green
}

# Apply to Python files
Write-Host "Applying headers to Python files..." -ForegroundColor Cyan
Get-ChildItem -Path . -Filter "*.py" -Recurse | ForEach-Object {
    Add-Header -FilePath $_.FullName -Header $HEADER_PYTHON
}

# Apply to JavaScript/TypeScript files
Write-Host "Applying headers to JavaScript/TypeScript files..." -ForegroundColor Cyan
Get-ChildItem -Path . -Include "*.js","*.ts","*.jsx","*.tsx" -Recurse | ForEach-Object {
    Add-Header -FilePath $_.FullName -Header $HEADER_JS
}

Write-Host "Header application complete!" -ForegroundColor Green
